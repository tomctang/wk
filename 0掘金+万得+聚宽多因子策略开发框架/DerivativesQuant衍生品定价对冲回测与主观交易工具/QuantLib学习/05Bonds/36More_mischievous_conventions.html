import QuantLib as ql
import pandas as pd

today = ql.Date(27, ql.January, 2011)
ql.Settings.instance().evaluationDate = today

issueDate = ql.Date(28, ql.January, 2011)
maturity = ql.Date(31, ql.August, 2020)
schedule = ql.Schedule(issueDate, maturity, ql.Period(ql.Semiannual), ql.UnitedStates(ql.UnitedStates.GovernmentBond), ql.Unadjusted, ql.Unadjusted, ql.DateGeneration.Backward, False)

bond = ql.FixedRateBond(1, 100.0, schedule, [0.03625], ql.ActualActual(ql.ActualActual.Bond), ql.Unadjusted, 100.0)

bond_yield = 0.034921
P1 = bond.dirtyPrice(bond_yield, bond.dayCounter(), ql.Compounded, ql.Semiannual)

engine = ql.DiscountingBondEngine(ql.YieldTermStructureHandle(flat_curve))
bond.setPricingEngine(engine)
